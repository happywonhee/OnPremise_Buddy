---
- name: Deploy and run test script
  hosts: all
  tasks:
    - name: Copy test script to remote host
      ansible.builtin.copy:
        src: ~/test.sh
        dest: /root/test.sh
        mode: 0755

    - name: Execute test script on remote host
      ansible.builtin.command:
        cmd: ~/test.sh
      register: script_output

    - name: Display script output
      ansible.builtin.debug:
        var: script_output.stdout_lines
